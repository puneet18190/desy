  def self.log(env, exception)
    LOGGER.error <<-LOG

-- BEG ENV --
#{env.inspect}
-- END ENV --
-- BEG EXCEPTION (#{exception.class}) --
message:    #{exception.message}
backtrace:  #{exception.backtrace.join("\n")}
-- END EXCEPTION (#{exception.class}) --
LOG
  end
